#define DITHERING TRUE

#define cr vec3(0.918, 0.265, 0.210)
#define cb vec3(0.260, 0.520, 0.955)
#define cy vec3(0.982, 0.739, 0.020)
#define cg vec3(0.045, 0.617, 0.347)

/* http://www.iryoku.com/next-generation-post-processing-in-call-of-duty-advanced-warfare */
float gradientNoise(in vec2 uv)
{
	return fract(52.9829189 * fract(dot(uv, vec2(0.06711056, 0.00583715))));
}

float flare(vec2 uv, vec2 pos) {
    // Transform uv for distortion of the blob
    vec2 cuv = (log2(uv+1.0) - pos)*2.;
    // Dot between uv itself is distance, creates a circle 
    return 1.3 - dot(cuv, cuv);
}

void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
    vec2 uv = (2. * fragCoord - iResolution.xy) / iResolution.y;
    float t = iTime*0.222+15.;
    // Scale linearly until good zoom level
    uv /= min(((iTime/5.)-8.)/5.+2.0, 7.0);
    vec3 col = vec3(0.2, 0.17, 0.19);

    col = mix(col, cr, flare(uv, vec2(-0.4, 0.0) * sin(t-45.)));
    col = mix(col, cb, flare(uv, vec2(-0.5, 0.2) * sin(-t*0.666) * cos(-t+33.)));
    col = mix(col, cy, flare(uv, vec2(0.13, -0.69) * sin(t-2.)));
    col = mix(col, cg, flare(uv, vec2(-0.5, 0.9) * sin(t)*cos(t/2.)));

    // Vignnete to avoid overexpossure
    col -= vec3(1.) * dot(uv, uv);
    
    // Apply dithering if needed
    #ifdef DITHERING
    col += (1.0 / 255.0) * gradientNoise(gl_FragCoord.xy) - (0.5 / 255.0);
    #endif

    // Output to screen
    fragColor = vec4(col, 1.0);
}

float stripes(float x, float f)
{
  float t = .5 + .5 * sin(f * 2. * PI * x);
  return t * t - .5;
}

void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
    vec2 uv = fragCoord/iResolution.xy;
    vec2 ouv = uv;
    float t = iTime+1.;
    t *= 0.1;

    uv += vec2(3.0) + vec2(sin(t/2.), cos(t/2.));

    vec3 col = vec3(bg);
    float a = sin(sin(ouv.x) + cos(ouv.y*3.) + uv.x*sqrt(log(uv.y)))*uv.y;
    col = mix(c1, col, stripes(a, cos(uv.x)*10.));

    fragColor = vec4(col,1.0);
}

#include <GL/glew.h>
#include <GLFW/glfw3.h>
#include <iostream>

// Definição inline dos shaders
const char* vertexShaderSource = R"(
    #version 330 core
    layout (location = 0) in vec3 aPos;
    
    void main() {
        gl_Position = vec4(aPos.x, aPos.y, aPos.z, 1.0);
    }
)";

const char* fragmentShaderSource = R"(
    #version 330 core
    out vec4 FragColor;
    
    void main() {
        FragColor = vec4(1.0f, 0.5f, 0.2f, 1.0f); // Cor laranja
    }
)";

int main() {
    // Inicializa GLFW
    if (!glfwInit()) {
        std::cerr << "Falha ao inicializar GLFW" << std::endl;
        return -1;
    }

    // Configura versão do OpenGL e perfil
    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
    glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);

    // Cria uma janela
    GLFWwindow* window = glfwCreateWindow(800, 600, "Triângulo OpenGL", NULL, NULL);
    if (window == NULL) {
        std::cerr << "Falha ao criar janela GLFW" << std::endl;
        glfwTerminate();
        return -1;
    }
    
    // Torna o contexto da janela atual
    glfwMakeContextCurrent(window);

    // Inicializa GLEW
    glewExperimental = GL_TRUE;
    if (glewInit() != GLEW_OK) {
        std::cerr << "Falha ao inicializar GLEW" << std::endl;
        return -1;
    }

    // Compila o shader de vértice
    unsigned int vertexShader = glCreateShader(GL_VERTEX_SHADER);
    glShaderSource(vertexShader, 1, &vertexShaderSource, NULL);
    glCompileShader(vertexShader);

    // Verifica erros de compilação - vertex shader
    int success;
    char infoLog[512];
    glGetShaderiv(vertexShader, GL_COMPILE_STATUS, &success);
    if (!success) {
        glGetShaderInfoLog(vertexShader, 512, NULL, infoLog);
        std::cerr << "ERRO::SHADER::VERTEX::COMPILACAO_FALHOU\n" << infoLog << std::endl;
    }

    // Compila o shader de fragmento
    unsigned int fragmentShader = glCreateShader(GL_FRAGMENT_SHADER);
    glShaderSource(fragmentShader, 1, &fragmentShaderSource, NULL);
    glCompileShader(fragmentShader);

    // Verifica erros de compilação - fragment shader
    glGetShaderiv(fragmentShader, GL_COMPILE_STATUS, &success);
    if (!success) {
        glGetShaderInfoLog(fragmentShader, 512, NULL, infoLog);
        std::cerr << "ERRO::SHADER::FRAGMENT::COMPILACAO_FALHOU\n" << infoLog << std::endl;
    }

    // Cria e liga o programa de shader
    unsigned int shaderProgram = glCreateProgram();
    glAttachShader(shaderProgram, vertexShader);
    glAttachShader(shaderProgram, fragmentShader);
    glLinkProgram(shaderProgram);

    // Verifica erros de ligação
    glGetProgramiv(shaderProgram, GL_LINK_STATUS, &success);
    if (!success) {
        glGetProgramInfoLog(shaderProgram, 512, NULL, infoLog);
        std::cerr << "ERRO::PROGRAMA::LIGACAO_FALHOU\n" << infoLog << std::endl;
    }

    // Libera os shaders após ligação
    glDeleteShader(vertexShader);
    glDeleteShader(fragmentShader);

    // Define os vértices do triângulo
    float vertices[] = {
        -0.5f, -0.5f, 0.0f, // Inferior esquerdo
         0.5f, -0.5f, 0.0f, // Inferior direito
         0.0f,  0.5f, 0.0f  // Superior meio
    };

    // Cria objetos de buffer
    unsigned int VAO, VBO;
    glGenVertexArrays(1, &VAO);
    glGenBuffers(1, &VBO);
    
    // Vincula o VAO primeiro
    glBindVertexArray(VAO);
    
    // Vincula e configura o VBO
    glBindBuffer(GL_ARRAY_BUFFER, VBO);
    glBufferData(GL_ARRAY_BUFFER, sizeof(vertices), vertices, GL_STATIC_DRAW);
    
    // Configura atributo de vértice
    glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 3 * sizeof(float), (void*)0);
    glEnableVertexAttribArray(0);
    
    // Desvincula VBO e VAO
    glBindBuffer(GL_ARRAY_BUFFER, 0);
    glBindVertexArray(0);

    // Loop de renderização
    while (!glfwWindowShouldClose(window)) {
        // Processa entrada do usuário
        if (glfwGetKey(window, GLFW_KEY_ESCAPE) == GLFW_PRESS)
            glfwSetWindowShouldClose(window, true);
        
        // Limpa a tela
        glClearColor(0.2f, 0.3f, 0.3f, 1.0f); // Cor de fundo verde-azulado escuro
        glClear(GL_COLOR_BUFFER_BIT);
        
        // Desenha o triângulo
        glUseProgram(shaderProgram);
        glBindVertexArray(VAO);
        glDrawArrays(GL_TRIANGLES, 0, 3);
        
        // Troca os buffers e verifica eventos
        glfwSwapBuffers(window);
        glfwPollEvents();
    }

    // Libera recursos
    glDeleteVertexArrays(1, &VAO);
    glDeleteBuffers(1, &VBO);
    glDeleteProgram(shaderProgram);
    
    // Finaliza GLFW
    glfwTerminate();
    return 0;
}
